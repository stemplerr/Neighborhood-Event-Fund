/****** Object:  StoredProcedure [dbo].[GetSimchosWithContributorCount]    Script Date: 8/12/2016 1:26:15 PM ******/
SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
CREATE PROCEDURE [dbo].[GetSimchosWithContributorCount]
AS
BEGIN
select COUNT(c.id) as contrcount, s.BaalSimcha, s.Date, s.total
from simchos s
left join ManyToMany m 
on s.id = m.SimchaId
left join Contributors c
on c.Id = m.ContributorId
group by s.BaalSimcha, s.Date, s.total
END
GO
